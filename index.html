Frontend Development

Responsive Design Essentials

E-commerce sites must perform flawlessly across all devices, particularly mobile, where an increasing share of online shopping takes place.

Key Responsive Design Principles:

  • Mobile-First Approach: Design for mobile, then enhance for larger screens
  • Fluid Grids: Use relative units (%, rem) instead of fixed pixels
  • Flexible Images: Ensure images scale properly across devices
  • Media Queries: Apply different styles based on device characteristics
  • Touch-Friendly: Larger tap targets for mobile users (min 44px)
  • Performance Focus: Optimize for speed on mobile networks

Responsive CSS Example:

/* Base styles for all devices */
.product-grid {
  display: grid;
  grid-gap: 1rem;
  margin-bottom: 2rem;
}

/* Mobile first - single column by default */
.product-card {
  width: 100%;
  border: 1px solid #eee;
  border-radius: 0.25rem;
  padding: 1rem;
}

/* Tablet styles */
@media (min-width: 768px) {

  }

  }

Backend Implementation

Database Design

A well-structured database is crucial for e-commerce performance and scalability.

Core Database Tables

Table Purpose Key Fields
products Store product information id, name, description, price, stock
categories Organize products id, name, parent_id
users Store customer information id, name, email, password, address
orders Track customer orders id, user_id, total, status, date
order_items Items within orders id, order_id, product_id, quantity, price
reviews Product reviews and ratings id, product_id, user_id, rating, comment

Database Relationships

Key Relationships:
  • One-to-Many: A category has many products
  • One-to-Many: A user has many orders
  • One-to-Many: An order has many order items
  • Many-to-Many: Products can belong to multiple categories (using junction table)

SQL Schema Example:

-- Products Table
CREATE TABLE products (
  id INT PRIMARY KEY AUTO_INCREMENT,
  name VARCHAR(255) NOT NULL,
  description TEXT,
  price DECIMAL(10, 2) NOT NULL,
  compare_price DECIMAL(10, 2),
  cost DECIMAL(10, 2),
  sku VARCHAR(100) UNIQUE,
  barcode VARCHAR(100),
  weight DECIMAL(8, 2),
  stock_quantity INT NOT NULL DEFAULT 0,
  low_stock_threshold INT DEFAULT 5,
  status ENUM('active', 'draft', 'archived') DEFAULT 'draft',
  featured BOOLEAN DEFAULT FALSE,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

-- Categories Table
CREATE TABLE categories (
  id INT PRIMARY KEY AUTO_INCREMENT,
  name VARCHAR(255) NOT NULL,
  slug VARCHAR(255) NOT NULL UNIQUE,
  description TEXT,
  parent_id INT,
  FOREIGN KEY (parent_id) REFERENCES categories(id) ON DELETE SET NULL
);

-- Product Categories (Many-to-Many)
CREATE TABLE product_categories (
  product_id INT,
  category_id INT,
  PRIMARY KEY (product_id, category_id),
  FOREIGN KEY (product_id) REFERENCES products(id) ON DELETE CASCADE,
  FOREIGN KEY (category_id) REFERENCES categories(id) ON DELETE CASCADE
);

-- Orders Table
CREATE TABLE orders (
  id INT PRIMARY KEY AUTO_INCREMENT,
  user_id INT,
  status ENUM('pending', 'processing', 'shipped', 'delivered', 'cancelled') DEFAULT 'pending',
  total DECIMAL(10, 2) NOT NULL,
  subtotal DECIMAL(10, 2) NOT NULL,
  tax DECIMAL(10, 2) NOT NULL,
  shipping DECIMAL(10, 2) NOT NULL,
  discount DECIMAL(10, 2) DEFAULT 0,
  shipping_address JSON,
  billing_address JSON,
  payment_method VARCHAR(50),
  notes TEXT,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE SET NULL
);

-- Order Items Table
CREATE TABLE order_items (
  id INT PRIMARY KEY AUTO_INCREMENT,
  order_id INT NOT NULL,
  product_id INT NOT NULL,
  quantity INT NOT NULL,
  price DECIMAL(10, 2) NOT NULL,
  FOREIGN KEY (order_id) REFERENCES orders(id) ON DELETE CASCADE,
  FOREIGN KEY (product_id) REFERENCES products(id) ON DELETE CASCADE
);
Database Optimization Tips:
  • Use appropriate indexes on frequently queried columns
  • Consider database sharding for very large product catalogs
  • Implement caching for product and category data
  • Use database transactions for order processing
  • Regularly backup your database and test recovery procedures

API Implementation with Express.js

// api/products.js - Example Express.js API routes for products
const express = require('express');
const router = express.Router();
const pool = require('../database');
const { authenticate } = require('../middleware/auth');

// Get all products with pagination
router.get('/', async (req, res) => {
  try {
    const page = parseInt(req.query.page) || 1;
    const limit = parseInt(req.query.limit) || 10;
    const offset = (page - 1) * limit;
    const category = req.query.category;
    
    let query = 'SELECT * FROM products WHERE status = "active"';
    let countQuery = 'SELECT COUNT(*) as count FROM products WHERE status = "active"';
    
    // Add category filter if provided
    if (category) {
      query += ` AND id IN (SELECT product_id FROM product_categories WHERE category_id = ?)`;
      countQuery += ` AND id IN (SELECT product_id FROM product_categories WHERE category_id = ?)`;
    }
    
    // Add sorting
    const sortBy = req.query.sort_by || 'created_at';
    const sortOrder = req.query.sort_order === 'asc' ? 'ASC' : 'DESC';
    query += ` ORDER BY ${sortBy} ${sortOrder}`;
    
    // Add pagination
    query += ' LIMIT ? OFFSET ?';
    
    // Execute query
    const [products] = await pool.query(
      query, 
      category ? [category, limit, offset] : [limit, offset]
    );
    
    // Get total count for pagination
    const [countResult] = await pool.query(
      countQuery,
      category ? [category] : []
    );
    const totalProducts = countResult[0].count;
    
    res.json({
      data: products,
      meta: {
        total: totalProducts,
        page,
        limit,
        totalPages: Math.ceil(totalProducts / limit)
      }
    });
  } catch (error) {
    console.error('Error fetching products:', error);
    res.status(500).json({ error: 'Failed to fetch products' });
  }
});

// Get a specific product by ID
router.get('/:id', async (req, res) => {
  try {
    const [products] = await pool.query(
      'SELECT * FROM products WHERE id = ?',
      [req.params.id]
    );
    
    if (products.length === 0) {
      return res.status(404).json({ error: 'Product not found' });
    }
    
    // Get product categories
    const [categories] = await pool.query(
      `SELECT c.id, c.name FROM categories c
       INNER JOIN product_categories pc ON c.id = pc.category_id
       WHERE pc.product_id = ?`,
      [req.params.id]
    );
    
    // Get product reviews
    const [reviews] = await pool.query(
      `SELECT r.*, u.name as user_name FROM reviews r
       LEFT JOIN users u ON r.user_id = u.id
       WHERE r.product_id = ?
       ORDER BY r.created_at DESC`,
      [req.params.id]
    );
    
    const product = {
      ...products[0],
      categories,
      reviews
    };
    
    res.json(product);
  } catch (error) {
    console.error('Error fetching product:', error);
    res.status(500).json({ error: 'Failed to fetch product' });
  }
});

// Create a new product (admin only)
router.post('/', authenticate, async (req, res) => {
  // Check if user is admin
  if (req.user.role !== 'admin') {
    return res.status(403).json({ error: 'Unauthorized' });
  }
  
  try {
    const {
      name, description, price, compare_price, cost,
      sku, barcode, weight, stock_quantity,
      low_stock_threshold, status, featured, 
      categories
    } = req.body;
    
    // Validate required fields
    if (!name || !price) {
      return res.status(400).json({ error: 'Name and price are required' });
    }
    
    // Insert product using transaction
    const connection = await pool.getConnection();
    await connection.beginTransaction();
    
    try {
      // Insert product
      const [result] = await connection.query(
        `INSERT INTO products (
          name, description, price, compare_price, cost,
          sku, barcode, weight, stock_quantity,
          low_stock_threshold, status, featured
        ) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)`,
        [
          name, description, price, compare_price, cost,
          sku, barcode, weight, stock_quantity,
          low_stock_threshold, status, featured
        ]
      );
      
      const productId = result.insertId;
      
      // Insert product categories if provided
      if (categories && categories.length > 0) {
        const categoryValues = categories.map(
          categoryId => [productId, categoryId]
        );
        
        await connection.query(
          `INSERT INTO product_categories (product_id, category_id) VALUES ?`,
          [categoryValues]
        );
      }
      
      await connection.commit();
      
      res.status(201).json({
        id: productId,
        message: 'Product created successfully'
      });
    } catch (error) {
      await connection.rollback();
      throw error;
    } finally {
      connection.release();
    }
  } catch (error) {
    console.error('Error creating product:', error);
    res.status(500).json({ error: 'Failed to create product' });
  }
});

module.exports = router;

Authentication & User Management

Implementing secure authentication is critical for e-commerce platforms to protect user data and manage access to resources.

Authentication Implementation

// auth.js - Authentication routes and middleware
const express = require('express');
const router = express.Router();
const bcrypt = require('bcrypt');
const jwt = require('jsonwebtoken');
const pool = require('../database');

// Register a new user
router.post('/register', async (req, res) => {
  try {
    const { name, email, password } = req.body;
    
    // Validate input
    if (!name || !email || !password) {
      return res.status(400).json({ error: 'All fields are required' });
    }
    
    // Check if email already exists
    const [users] = await pool.query(
      'SELECT id FROM users WHERE email = ?', 
      [email]
    );
    
    if (users.length > 0) {
      return res.status(400).json({ error: 'Email already in use' });
    }
    
    // Hash password
    const salt = await bcrypt.genSalt(10);
    const hashedPassword = await bcrypt.hash(password, salt);
    
    // Insert user
    const [result] = await pool.query(
      'INSERT INTO users (name, email, password, role) VALUES (?, ?, ?, ?)',
      [name, email, hashedPassword, 'customer']
    );
    
    // Generate JWT token
    const token = jwt.sign(
      { id: result.insertId, email, role: 'customer' },
      process.env.JWT_SECRET,
      { expiresIn: '1h' }
    );
    
    res.status(201).json({
      message: 'Registration successful',
      token,
      user: {
        id: result.insertId,
        name,
        email,
        role: 'customer'
      }
    });
  } catch (error) {
    console.error('Registration error:', error);
    res.status(500).json({ error: 'Registration failed' });
  }
});

// Login
router.post('/login', async (req, res) => {
  try {
    const { email, password } = req.body;
    
    // Validate input
    if (!email || !password) {
      return res.status(400).json({ error: 'Email and password are required' });
    }
    
    // Find user
    const [users] = await pool.query(
      'SELECT id, name, email, password, role FROM users WHERE email = ?',
      [email]
    );
    
    if (users.length === 0) {
      return res.status(400).json({ error: 'Invalid credentials' });
    }
    
    const user = users[0];
    
    // Check password
    const isMatch = await bcrypt.compare(password, user.password);
    
    if (!isMatch) {
      return res.status(400).json({ error: 'Invalid credentials' });
    }
    
    // Generate JWT token
    const token = jwt.sign(
      { id: user.id, email: user.email, role: user.role },
      process.env.JWT_SECRET,
      { expiresIn: '1h' }
    );
    
    res.json({
      message: 'Login successful',
      token,
      user: {
        id: user.id,
        name: user.name,
        email: user.email,
        role: user.role
      }
    });
  } catch (error) {
    console.error('Login error:', error);
    res.status(500).json({ error: 'Login failed' });
  }
});

// Authentication middleware
const authenticate = (req, res, next) => {
  // Get token from header
  const token = req.header('Authorization')?.split(' ')[1];
  
  if (!token) {
    return res.status(401).json({ error: 'No token, authorization denied' });
  }
  
  try {
    // Verify token
    const decoded = jwt.verify(token, process.env.JWT_SECRET);
    
    // Add user to request
    req.user = decoded;
    next();
  } catch (error) {
    console.error('Token verification error:', error);
    res.status(401).json({ error: 'Token is not valid' });
  }
};

module.exports = { router, authenticate };

User Profile Management

// users.js - User profile management routes
const express = require('express');
const router = express.Router();
const bcrypt = require('bcrypt');
const pool = require('../database');
const { authenticate } = require('../middleware/auth');

// Get current user profile
router.get('/profile', authenticate, async (req, res) => {
  try {
    const [users] = await pool.query(
      `SELECT id, name, email, phone, role, created_at 
       FROM users WHERE id = ?`,
      [req.user.id]
    );
    
    if (users.length === 0) {
      return res.status(404).json({ error: 'User not found' });
    }
    
    // Get user addresses
    const [addresses] = await pool.query(
      'SELECT * FROM addresses WHERE user_id = ?',
      [req.user.id]
    );
    
    // Get user's recent orders
    const [orders] = await pool.query(
      `SELECT id, status, total, created_at 
       FROM orders 
       WHERE user_id = ? 
       ORDER BY created_at DESC 
       LIMIT 5`,
      [req.user.id]
    );
    
    const user = {
      ...users[0],
      addresses,
      recentOrders: orders
    };
    
    res.json(user);
  } catch (error) {
    console.error('Error fetching profile:', error);
    res.status(500).json({ error: 'Failed to fetch profile' });
  }
});

// Update user profile
router.put('/profile', authenticate, async (req, res) => {
  try {
    const { name, email, phone } = req.body;
    
    // Validate input
    if (!name || !email) {
      return res.status(400).json({ error: 'Name and email are required' });
    }
    
    // Check if email is already taken by another user
    if (email !== req.user.email) {
      const [users] = await pool.query(
        'SELECT id FROM users WHERE email = ? AND id != ?',
        [email, req.user.id]
      );
      
      if (users.length > 0) {
        return res.status(400).json({ error: 'Email already in use' });
      }
    }
    
    // Update user
    await pool.query(
      'UPDATE users SET name = ?, email = ?, phone = ? WHERE id = ?',
      [name, email, phone, req.user.id]
    );
    
    res.json({ message: 'Profile updated successfully' });
  } catch (error) {
    console.error('Error updating profile:', error);
    res.status(500).json({ error: 'Failed to update profile' });
  }
});

      [hashedPassword, req.user.id]
    );
    
    res.json({ message: 'Password changed successfully' });
  } catch (error) {

// inventory.js - Inventory management functions
const pool = require('../database');

// Update inventory when order is placed
async function updateInventory(items, connection) {
  for (const item of items) {
    // Check if product exists and has sufficient stock
    const [products] = await connection.query(
      'SELECT id, stock_quantity FROM products WHERE id = ?',
      [item.productId]
    );
    
    if (products.length === 0) {
      throw new Error(`Product with ID ${item.productId} not found`);
    }
    
    const product = products[0];
    
    // Check if product variation exists
    if (item.variationId) {
      const [variations] = await connection.query(
        'SELECT id, stock_quantity FROM product_variations WHERE id = ?',
        [item.variationId]
      );
      
      if (variations.length === 0) {
        throw new Error(`Variation with ID ${item.variationId} not found`);
      }
      
      const variation = variations[0];
      
      if (variation.stock_quantity < item.quantity) {
        throw new Error(`Insufficient stock for variation ${item.variationId}`);
      }
      
      // Update variation stock
      await connection.query(
        'UPDATE product_variations SET stock_quantity = stock_quantity - ? WHERE id = ?',
        [item.quantity, item.variationId]
      );
    } else {
      // Check if product has sufficient stock
      if (product.stock_quantity < item.quantity) {
        throw new Error(`Insufficient stock for product ${item.productId}`);
      }
      
      // Update product stock
      await connection.query(
        'UPDATE products SET stock_quantity = stock_quantity - ? WHERE id = ?',
        [item.quantity, item.productId]
      );
    }
  }
}

// Check if product is in stock
async function checkStock(productId, variationId = null, quantity = 1) {
  try {
    if (variationId) {
      const [variations] = await pool.query(
        'SELECT stock_quantity FROM product_variations WHERE id = ?',
        [variationId]
      );
      
      if (variations.length === 0 || variations[0].stock_quantity < quantity) {
        return false;
      }
    } else {
      const [products] = await pool.query(
        'SELECT stock_quantity FROM products WHERE id = ?',
        [productId]
      );
      
      if (products.length === 0 || products[0].stock_quantity < quantity) {
        return false;
      }
    }
    
    return true;
  } catch (error) {
    console.error('Error checking stock:', error);
    return false;
  }
}

// filtering.js - Dynamic product filtering 
const pool = require('../database');

// Get filtered products
async function getFilteredProducts(options = {}) {
  try {
    const {
      category,
      price_min,
      price_max,
      attributes = {},
      sort_by = 'created_at',
      sort_order = 'desc',
      page = 1,
      limit = 20
    } = options;
    
    const offset = (page - 1) * limit;
    const params = [];
    
    // Start building the query
    let query = `
      SELECT DISTINCT p.* 
      FROM products p
    `;
    
    // Join tables based on filters
    if (category) {
      query += `
        INNER JOIN product_categories pc ON p.id = pc.product_id
      `;
    }
    
    // Add attribute joins if needed
    if (Object.keys(attributes).length > 0) {
      query += `
        INNER JOIN product_variations pv ON p.id = pv.product_id
        INNER JOIN variation_attribute_values vav ON pv.id = vav.variation_id
      `;
    }
    
    // Start the WHERE clause
    query += ` WHERE p.status = 'active'`;
    
    // Add category filter
    if (category) {
      query += ` AND pc.category_id = ?`;
      params.push(category);
    }
    
    // Add price range filter
    if (price_min) {
      query += ` AND p.price >= ?`;
      params.push(price_min);
    }
    
    if (price_max) {
      query += ` AND p.price <= ?`;
      params.push(price_max);
    }
    
    // Add attribute filters
    Object.entries(attributes).forEach(([attribute, values]) => {
      if (values && values.length > 0) {
        const attributeParams = values.map(v => '?').join(',');
        query += `
          AND vav.attribute_value_id IN (
            SELECT av.id 
            FROM attribute_values av
            INNER JOIN product_attributes pa ON av.attribute_id = pa.id
            WHERE pa.name = ? AND av.value IN (${attributeParams})
          )
        `;
        params.push(attribute, ...values);
      }
    });
    
    // Count total matching products
    const countQuery = `SELECT COUNT(DISTINCT p.id) as total FROM ${query.split('FROM')[1]}`;
    const [countResult] = await pool.query(countQuery, params);
    const totalProducts = countResult[0].total;
    
    // Add sorting
    query += ` ORDER BY p.${sort_by} ${sort_order === 'asc' ? 'ASC' : 'DESC'}`;
    
    // Add pagination
    query += ` LIMIT ? OFFSET ?`;
    params.push(limit, offset);
    
    // Execute the final query
    const [products] = await pool.query(query, params);
    
    // Get available filters based on matching products
    const filters = await getAvailableFilters(options);
    
    return {
      products,
      filters,
      pagination: {
        total: totalProducts,
        page,
        limit,
        totalPages: Math.ceil(totalProducts / limit)
      }
    };
  } catch (error) {
    console.error('Error getting filtered products:', error);
    throw error;
  }
}

// Get available filters based on the current result set
async function getAvailableFilters(options) {
  try {
    const { category } = options;
    const filters = {};
    
    // Build base query for products matching the current filters
    let baseQuery = `
      SELECT DISTINCT p.id 
      FROM products p
    `;
    
    if (category) {
      baseQuery += `
        INNER JOIN product_categories pc ON p.id = pc.product_id
        WHERE pc.category_id = ?
      `;
    } else {
      baseQuery += ` WHERE 1=1`;
    }
    
    const baseParams = category ? [category] : [];
    
    // Get price range
    const [priceRange] = await pool.query(`
      SELECT MIN(price) as min_price, MAX(price) as max_price
      FROM products p
      ${category ? 'INNER JOIN product_categories pc ON p.id = pc.product_id' : ''}
      WHERE p.status = 'active' ${category ? 'AND pc.category_id = ?' : ''}
    `, baseParams);
    
    filters.price_range = {
      min: priceRange[0].min_price,
      max: priceRange[0].max_price
    };
    
    // Get attribute filters
    const [attributes] = await pool.query(`
      SELECT 
        pa.id, pa.name, pa.display_name, pa.type,
        av.id as value_id, av.value, av.display_value,
        COUNT(DISTINCT pv.product_id) as product_count
      FROM 
        product_attributes pa
        INNER JOIN attribute_values av ON pa.id = av.attribute_id
        INNER JOIN variation_attribute_values vav ON av.id = vav.attribute_value_id
        INNER JOIN product_variations pv ON vav.variation_id = pv.id
        INNER JOIN products p ON pv.product_id = p.id
        ${category ? 'INNER JOIN product_categories pc ON p.id = pc.product_id' : ''}
      WHERE 
        p.status = 'active'
        ${category ? 'AND pc.category_id = ?' : ''}
      GROUP BY 
        pa.id, av.id
      ORDER BY 
        pa.display_name, av.display_value
    `, baseParams);
    
    // Organize attributes by name
    filters.attributes = {};
    
    attributes.forEach(attr => {
      if (!filters.attributes[attr.name]) {
        filters.attributes[attr.name] = {
          id: attr.id,
          name: attr.name,
          display_name: attr.display_name,
          type: attr.type,
          values: []
        };
      }
      
      filters.attributes[attr.name].values.push({
        id: attr.value_id,
        value: attr.value,
        display_value: attr.display_value,
        product_count: attr.product_count
      });
    });
    
    return filters;
  } catch (error) {
    console.error('Error getting available filters:', error);
    throw error;
  }
}

Shopping Cart Implementation

A well-designed shopping cart is critical for conversion and provides a smooth path to checkout.

Cart JavaScript Implementation

// cart.js - Shopping cart functionality
$(document).ready(function() {
  // Cart object to manage products
  const cart = {
    items: [],
    subtotal: 0,
    tax: 0,
    shipping: 0,
    total: 0,
    
    // Initialize cart from localStorage or session
    init: function() {
      // Try to load from localStorage
      const savedCart = localStorage.getItem('cart');
      
      if (savedCart) {
        try {
          const parsedCart = JSON.parse(savedCart);
          this.items = parsedCart.items || [];
          this.recalculate();
        } catch (e) {
          console.error('Error parsing saved cart:', e);
          this.clearCart();
        }
      }
      
      this.renderCart();
      this.updateCartCount();
    },
    
    // Add item to cart
    addItem: function(product) {
      // Check if product already exists in cart
      const existingItem = this.items.find(item => 
        item.productId === product.id && 
        item.variationId === (product.variationId || null)
      );
      
      if (existingItem) {
        // Update quantity if already in cart
        existingItem.quantity += product.quantity || 1;
      } else {
        // Add new item to cart
        this.items.push({
          productId: product.id,
          variationId: product.variationId || null,
          name: product.name,
          price: product.price,
          salePrice: product.salePrice || product.price,
          image: product.image,
          quantity: product.quantity || 1,
          attributes: product.attributes || {}
        });
      }
      
      this.saveCart();
      this.recalculate();
      this.renderCart();
      this.updateCartCount();
      this.showCartNotification(product.name);
    },
    
    // Update item quantity
    updateQuantity: function(index, quantity) {
      if (index >= 0 && index < this.items.length) {
        if (quantity <= 0) {
          // Remove item if quantity is 0 or negative
          this.removeItem(index);
        } else {
          // Update quantity
          this.items[index].quantity = quantity;
          this.saveCart();
          this.recalculate();
          this.renderCart();
        }
      }
    },
    
    // Remove item from cart
    removeItem: function(index) {
      if (index >= 0 && index < this.items.length) {
        this.items.splice(index, 1);
        this.saveCart();
        this.recalculate();
        this.renderCart();
        this.updateCartCount();
      }
    },
    
    // Clear entire cart
    clearCart: function() {
      this.items = [];
      this.saveCart();
      this.recalculate();
      this.renderCart();
      this.updateCartCount();
    },
    
    // Recalculate cart totals
    recalculate: function() {
      // Calculate subtotal
      this.subtotal = this.items.reduce((total, item) => {
        return total + (item.salePrice * item.quantity);
      }, 0);
      
      // Calculate tax (example: 8%)
      this.tax = this.subtotal * 0.08;

// cart-api.js - Server-side cart handling with Express
const express = require('express');

// Add item to cart
router.post('/items', async (req, res) => {
  try {
    const { productId, variationId, quantity = 1 } = req.body;
    
    // Validate required fields
    if (!productId) {
      return res.status(400).json({ error: 'Product ID is required' });
    }
    
    // Validate product exists
    const [products] = await pool.query(
      'SELECT * FROM products WHERE id = ? AND status = "active"',
      [productId]
    );
    
    if (products.length === 0) {
      return res.status(404).json({ error: 'Product not found' });
    }
    
    const product = products[0];
    
    // Check product stock
    if (product.stock_quantity < quantity) {
      return res.status(400).json({ error: 'Insufficient stock' });
    }
    
    // If variation provided, validate and get details
    let variation = null;
    if (variationId) {
      const [variations] = await pool.query(
        'SELECT * FROM product_variations WHERE id = ? AND product_id = ?',
        [variationId, productId]
      );
      
      if (variations.length === 0) {
        return res.status(404).json({ error: 'Variation not found' });
      }
      
      variation = variations[0];
      
      // Check variation stock
      if (variation.stock_quantity < quantity) {
        return res.status(400).json({ error: 'Insufficient stock for selected variation' });
      }
    }
    
    // Get product image
    const [images] = await pool.query(
      'SELECT url FROM product_media WHERE product_id = ? AND position = 1 LIMIT 1',
      [productId]
    );
    
    const imageUrl = images.length > 0 ? images[0].url : null;
    
    // Create cart item
    const cartItem = {
      productId,
      variationId: variationId || null,
      name: product.name,
      price: product.price,
      salePrice: product.sale_price || product.price,
      image: imageUrl,
      quantity
    };
    
    // If variation exists, get attributes
    if (variation) {
      // Get variation attributes
      const [attributes] = await pool.query(`
        SELECT 
          pa.name, av.value, av.display_value
        FROM 
          variation_attribute_values vav
          INNER JOIN attribute_values av ON vav.attribute_value_id = av.id
          INNER JOIN product_attributes pa ON av.attribute_id = pa.id
        WHERE 
          vav.variation_id = ?
      `, [variationId]);
      
      cartItem.attributes = attributes.reduce((acc, attr) => {
        acc[attr.name] = attr.display_value;
        return acc;
      }, {});
      
      // Use variation price if available
      if (variation.price) {
        cartItem.price = variation.price;
        cartItem.salePrice = variation.sale_price || variation.price;
      }
    }
    
    let cart;
    
    if (req.user) {
      // Logged in user: update cart in database
      cart = await addToUserCart(req.user.id, cartItem);
    } else {
      // Guest user: update cart in session
      cart = req.session.cart || { items: [], totals: calculateTotals([]) };
      
      // Check if item already exists in cart
      const existingItemIndex = cart.items.findIndex(item => 
        item.productId === cartItem.productId && 
        item.variationId === cartItem.variationId
      );
      
      if (existingItemIndex !== -1) {
        // Update existing item
        cart.items[existingItemIndex].quantity += cartItem.quantity;
      } else {
        // Add new item
        cart.items.push(cartItem);
      }
      
      // Recalculate totals
      cart.totals = calculateTotals(cart.items);
      
      // Save updated cart to session
      req.session.cart = cart;
    }
    
    res.json(cart);
  } catch (error) {
    console.error('Error adding to cart:', error);
    res.status(500).json({ error: 'Failed to add item to cart' });
  }
});

// Update cart item quantity
router.put('/items/:itemId', async (req, res) => {
  try {
    const { itemId } = req.params;
    const { quantity } = req.body;
    
    // Validate quantity
    if (!quantity || quantity < 0) {
      return res.status(400).json({ error: 'Invalid quantity' });
    }
    
    let cart;
    
    if (req.user) {
      // Logged in user: update cart in database
      if (quantity === 0) {
        // Remove item
        cart = await removeFromUserCart(req.user.id, itemId);
      } else {
        // Update quantity
        cart = await updateUserCartItemQuantity(req.user.id, itemId, quantity);
      }
    } else {
      // Guest user: update cart in session
      cart = req.session.cart || { items: [], totals: calculateTotals([]) };
      
      const itemIndex = parseInt(itemId);
      
      if (isNaN(itemIndex) || itemIndex < 0 || itemIndex >= cart.items.length) {
        return res.status(404).json({ error: 'Cart item not found' });
      }
      
      if (quantity === 0) {
        // Remove item
        cart.items.splice(itemIndex, 1);
      } else {
        // Update quantity
        cart.items[itemIndex].quantity = quantity;
      }
      
      // Recalculate totals
      cart.totals = calculateTotals(cart.items);
      
      // Save updated cart to session
      req.session.cart = cart;
    }
    
    res.json(cart);
  } catch (error) {
    console.error('Error updating cart item:', error);
    res.status(500).json({ error: 'Failed to update cart item' });
  }
});

// Helper function to calculate cart totals
function calculateTotals(items) {
  // Calculate subtotal
  const subtotal = items.reduce((total, item) => {
    return total + (item.salePrice * item.quantity);
  }, 0);
  
  // Calculate tax (example: 8%)
  const tax = subtotal * 0.08;
  
  // Calculate shipping (example: free over $50, otherwise $5.99)
  const shipping = subtotal > 50 ? 0 : 5.99;
  
  // Calculate total
  const total = subtotal + tax + shipping;
  
  return {
    subtotal,
    tax,
    shipping,
    total
  };
}

// Database helper functions
async function getUserCart(userId) {
  // Get user's cart from database
  const [cartItems] = await pool.query(`
    SELECT 
      ci.id,
      ci.product_id as productId,
      ci.variation_id as variationId,
      p.name,
      ci.price,
      ci.sale_price as salePrice,
      pm.url as image,
      ci.quantity,
      ci.attributes
    FROM 
      cart_items ci
      INNER JOIN products p ON ci.product_id = p.id
      LEFT JOIN product_media pm ON p.id = pm.product_id AND pm.position = 1
    WHERE 
      ci.user_id = ?
    ORDER BY 
      ci.created_at
  `, [userId]);
  
  // Parse attributes JSON
  const items = cartItems.map(item => ({
    ...item,
    attributes: item.attributes ? JSON.parse(item.attributes) : {}
  }));
  
  // Calculate totals
  const totals = calculateTotals(items);
  
  return { items, totals };
}

async function addToUserCart(userId, cartItem) {
  // Start transaction
  const connection = await pool.getConnection();
  await connection.beginTransaction();
  
  try {
    // Check if item already exists in cart
    const [existingItems] = await connection.query(`
      SELECT id, quantity 
      FROM cart_items 
      WHERE user_id = ? AND product_id = ? AND variation_id ${cartItem.variationId ? '= ?' : 'IS NULL'}
    `, cartItem.variationId 
      ? [userId, cartItem.productId, cartItem.variationId] 
      : [userId, cartItem.productId]
    );
    
    if (existingItems.length > 0) {
      // Update existing item
      await connection.query(`
        UPDATE cart_items
        SET quantity = quantity + ?, updated_at = NOW()
        WHERE id = ?
      `, [cartItem.quantity, existingItems[0].id]);
    } else {
      // Insert new item
      await connection.query(`
        INSERT INTO cart_items (
          user_id, product_id, variation_id, price, sale_price,
          quantity, attributes, created_at, updated_at
        ) VALUES (?, ?, ?, ?, ?, ?, ?, NOW(), NOW())
      `, [
        userId, 
        cartItem.productId, 
        cartItem.variationId, 
        cartItem.price,
        cartItem.salePrice,
        cartItem.quantity,
        JSON.stringify(cartItem.attributes || {})
      ]);
    }
    
    await connection.commit();
    
    // Get updated cart
    return await getUserCart(userId);
  } catch (error) {
    await connection.rollback();
    throw error;
  } finally {
    connection.release();
  }
}
